Функция ТекущийПользователь() Экспорт
	ТекушийПользователь = ПользователиИнформационнойБазы.ТекущийПользователь();
	ПользовательСсылка = Справочники.Пользователи.НайтиПоНаименованию(СокрЛП(ТекушийПользователь), Истина);
	Если НЕ ЗначениеЗаполнено(ПользовательСсылка) Тогда
		Пользователь = Справочники.Пользователи.СоздатьЭлемент();
		Пользователь.Наименование = СокрЛП(ТекушийПользователь);
		Пользователь.Записать();
		ПользовательСсылка = Пользователь.Ссылка;
	КонецЕсли;
	Возврат ПользовательСсылка.Ссылка;
КонецФункции

Функция ПолучитьНастройкуПользователя(Настройка)
	Если НЕ ЗначениеЗаполнено(глТекущийПользователь) Тогда
		Возврат Неопределено;
	КонецЕсли;
	РСНЗ = РегистрыСведений.НастройкиПользователей.СоздатьНаборЗаписей();
	РСНЗ.Отбор.Пользователь.Установить(глТекущийПользователь);
	РСНЗ.Прочитать();
	Если РСНЗ.Количество() = 0 Тогда
		ВОзврат Неопределено;
	КонецЕсли;
	ТаблицаНастроек = РСНЗ.Получить(0).Настройки.Получить();
	СтрокаНастройка = ТаблицаНастроек.Найти(Настройка, "Настройка");
	Если СтрокаНастройка = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	Возврат СтрокаНастройка.Значение;
КонецФункции
